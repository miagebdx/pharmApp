\newcommand*{\titleGM}{\begingroup % Create the command for including the title page in the document
\hbox{ % Horizontal box
\hspace*{0.2\textwidth} % Whitespace to the left of the title page
\rule{1pt}{\textheight} % Vertical line
\hspace*{0.05\textwidth} % Whitespace between the vertical line and title page text
\parbox[b]{0.75\textwidth}{ % Paragraph box which restricts text to less than the width of the page

{\noindent\Huge\bfseries PharmApp}\\[2\baselineskip] % Title
{\large \textit{Université de Bordeaux}}\\[4\baselineskip] % Tagline or further description
{\Large \textsc{Jérémy Morin \\ Louis Laine}} % Author name

\vspace{0.5\textheight} % Whitespace between the title block and the publisher
{\noindent }\\[\baselineskip] % Publisher and logo
}}
\endgroup}
